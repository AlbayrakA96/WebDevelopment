Maak opdracht 1, maar maak nu gebruik van een while loop.

            const sports = [
                "voetbal",
                "basketbal",
                "volleybal",
                "hockey",
                "honkbal",
                "zwemmen",
                "kortbal",
            ];        
        var text = "";
        let i = 0;
        while (i < sports.length) {
            text += sports[i] +" ";
            i++;
        }
        
        console.log(text);

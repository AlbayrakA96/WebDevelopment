Maak opdracht 1, maar maak nu gebruik van een for...in loop.

The for/in statement loops through the properties of an object.

            const sports = [
                "voetbal",
                "basketbal",
                "volleybal",
                "hockey",
                "honkbal",
                "zwemmen",
                "kortbal",
            ];  
            
            let x;
            
            for (x in sports) {
                console.log(sports[x]);
                
            }

            console.log(x);

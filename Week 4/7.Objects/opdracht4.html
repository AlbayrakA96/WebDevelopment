Sorteer persons op leeftijd van klein naar groot.

            const Person = function(name, age) {
                this.name = name;
                this.age = age;
                    }
            const persons = [
                    new Person("Jan", 24),
                    new Person("Klaas", 15),
                    new Person("Yanis", 42),
                    new Person("Rupel", 55),
                    new Person("Hendrik", 32),
                    new Person("Mono", 28),
                    new Person("West", 23),
                ]  
                 
                persons.sort((
                    function(a, b)
